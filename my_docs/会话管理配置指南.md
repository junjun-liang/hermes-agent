## 配置项详解

### 1. `session_timeout: int = 3600`

**作用：** 会话超时时间（秒）

```python
# 1 小时（默认）
session_timeout = 3600

# 30 分钟（短会话）
session_timeout = 1800

# 2 小时（长会话）
session_timeout = 7200

# 永不过期（不推荐）
session_timeout = -1
```

**配置说明：**
- 单位：秒
- 默认值：3600 秒（1 小时）
- 适用场景：用户无操作超过该时间，会话自动过期
- 安全建议：生产环境不宜设置过长

---

### 2. `max_sessions_per_user: int = 10`

**作用：** 每个用户最大并发会话数

```python
# 最多 10 个会话（默认）
max_sessions_per_user = 10

# 严格限制（只允许 1 个）
max_sessions_per_user = 1

# 宽松限制（允许多设备）
max_sessions_per_user = 50

# 无限制（不推荐）
max_sessions_per_user = -1
```

**配置说明：**
- 防止单用户占用过多资源
- 支持多设备登录场景
- 超出限制时，自动淘汰最旧会话
- 安全建议：根据业务需求合理设置

---

### 3. `session_db_path: Optional[str] = None`

**作用：** 会话数据库路径

```python
# None: 使用默认路径 ~/.hermes/state.db
session_db_path = None

# 自定义路径
session_db_path = "/var/lib/hermes/state.db"

# 内存数据库（测试用）
session_db_path = ":memory:"

# 分布式存储（Redis）
session_db_path = None  # 配合 redis_url 使用
```

**配置说明：**
- `None` 表示使用默认路径 `~/.hermes/state.db`
- 支持自定义 SQLite 数据库路径
- 生产环境建议使用 Redis 等分布式存储
- 测试环境可使用 `:memory:` 内存数据库

## 会话超时机制

### 超时策略

```python
# 1. 绝对超时（从创建开始计算）
class Session:
    def __init__(self, session_id: str, timeout: int):
        self.session_id = session_id
        self.created_at = datetime.now()
        self.expires_at = self.created_at + timedelta(seconds=timeout)
    
    def is_expired(self) -> bool:
        return datetime.now() > self.expires_at

# 2. 活动超时（从最后活动开始计算）
class ActivitySession(Session):
    def touch(self):
        """更新活动时间"""
        self.last_activity = datetime.now()
        self.expires_at = self.last_activity + timedelta(seconds=self.timeout)
    
    def is_expired(self) -> bool:
        return datetime.now() > self.expires_at
```

### 超时处理

```python
# 中间件实现会话超时检查
class SessionTimeoutMiddleware:
    async def dispatch(self, request: Request, call_next):
        session_id = request.headers.get("X-Session-ID")
        
        if session_id:
            session = await session_store.get(session_id)
            
            if session and session.is_expired():
                # 会话已过期
                await session_store.delete(session_id)
                raise HTTPException(401, "会话已过期")
            
            # 更新活动时间
            if session:
                session.touch()
                await session_store.update(session)
        
        return await call_next(request)
```

## 并发会话控制

### 会话数量限制

```python
class SessionStore:
    def __init__(self, max_sessions_per_user: int):
        self.max_sessions = max_sessions_per_user
        self.sessions = {}  # user_id -> [session_id, ...]
    
    async def create_session(self, user_id: str) -> str:
        # 检查当前会话数
        user_sessions = self.sessions.get(user_id, [])
        
        if len(user_sessions) >= self.max_sessions:
            # 超出限制，淘汰最旧会话
            oldest_session = user_sessions[0]
            await self.delete(oldest_session)
            user_sessions.pop(0)
        
        # 创建新会话
        session_id = generate_session_id()
        user_sessions.append(session_id)
        self.sessions[user_id] = user_sessions
        
        return session_id
```

### 并发访问控制

```python
import asyncio

class ConcurrentSessionStore(SessionStore):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.locks = {}  # session_id -> Lock
    
    async def get_lock(self, session_id: str) -> asyncio.Lock:
        """获取会话锁"""
        if session_id not in self.locks:
            self.locks[session_id] = asyncio.Lock()
        return self.locks[session_id]
    
    async def update_session(self, session_id: str, data: dict):
        """更新会话（加锁保护）"""
        lock = await self.get_lock(session_id)
        
        async with lock:
            # 临界区：防止并发写入
            session = await self.get(session_id)
            session.update(data)
            await self.save(session)
```

### 会话淘汰策略

```python
# 1. FIFO（先进先出）
def evict_oldest(sessions: list) -> str:
    return sessions.pop(0)  # 淘汰最旧

# 2. LRU（最近最少使用）
from collections import OrderedDict

class LRUSessionStore:
    def __init__(self, capacity: int):
        self.cache = OrderedDict()
        self.capacity = capacity
    
    async def access(self, session_id: str):
        """访问会话（移动到末尾）"""
        if session_id in self.cache:
            self.cache.move_to_end(session_id)
    
    async def evict(self):
        """淘汰最近最少使用的会话"""
        if len(self.cache) >= self.capacity:
            oldest = next(iter(self.cache))
            self.cache.pop(oldest)

# 3. LFU（最不经常使用）
class LFUSessionStore:
    def __init__(self, capacity: int):
        self.cache = {}  # session_id -> (data, frequency)
        self.capacity = capacity
    
    async def access(self, session_id: str):
        """增加访问频率"""
        if session_id in self.cache:
            data, freq = self.cache[session_id]
            self.cache[session_id] = (data, freq + 1)
    
    async def evict(self):
        """淘汰访问频率最低的会话"""
        if len(self.cache) >= self.capacity:
            # 找到频率最低的
            min_freq_session = min(self.cache.items(), key=lambda x: x[1][1])
            self.cache.pop(min_freq_session[0])
```

## 会话数据库配置

### SQLite 存储（默认）

```python
import sqlite3
from pathlib import Path

class SQLiteSessionStore:
    def __init__(self, db_path: str):
        self.db_path = Path(db_path)
        self.db_path.parent.mkdir(parents=True, exist_ok=True)
        self.init_db()
    
    def init_db(self):
        """初始化数据库"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # 创建会话表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS sessions (
                session_id TEXT PRIMARY KEY,
                user_id TEXT NOT NULL,
                messages TEXT NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                last_activity TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                metadata TEXT
            )
        """)
        
        # 创建索引
        cursor.execute("""
            CREATE INDEX IF NOT EXISTS idx_user_id 
            ON sessions(user_id)
        """)
        
        cursor.execute("""
            CREATE INDEX IF NOT EXISTS idx_last_activity 
            ON sessions(last_activity)
        """)
        
        conn.commit()
        conn.close()
    
    async def save(self, session: Session):
        """保存会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            INSERT OR REPLACE INTO sessions 
            (session_id, user_id, messages, last_activity, metadata)
            VALUES (?, ?, ?, ?, ?)
        """, (
            session.session_id,
            session.user_id,
            json.dumps(session.messages),
            datetime.now(),
            json.dumps(session.metadata),
        ))
        
        conn.commit()
        conn.close()
    
    async def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            SELECT * FROM sessions WHERE session_id = ?
        """, (session_id,))
        
        row = cursor.fetchone()
        conn.close()
        
        if row:
            return Session.from_row(row)
        return None
    
    async def delete(self, session_id: str):
        """删除会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cursor.execute("""
            DELETE FROM sessions WHERE session_id = ?
        """, (session_id,))
        
        conn.commit()
        conn.close()
    
    async def cleanup_expired(self, timeout_seconds: int):
        """清理过期会话"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        cutoff_time = datetime.now() - timedelta(seconds=timeout_seconds)
        
        cursor.execute("""
            DELETE FROM sessions 
            WHERE last_activity < ?
        """, (cutoff_time,))
        
        deleted_count = cursor.rowcount
        conn.commit()
        conn.close()
        
        return deleted_count
```

### Redis 存储（分布式）

```python
import redis.asyncio as redis
import json

class RedisSessionStore:
    def __init__(self, redis_url: str):
        self.redis = redis.from_url(
            redis_url,
            decode_responses=True
        )
    
    async def save(self, session: Session, ttl: int = 3600):
        """保存会话（设置 TTL）"""
        key = f"session:{session.session_id}"
        data = {
            "session_id": session.session_id,
            "user_id": session.user_id,
            "messages": session.messages,
            "metadata": session.metadata,
        }
        
        await self.redis.setex(
            key,
            ttl,  # 过期时间（秒）
            json.dumps(data)
        )
        
        # 维护用户会话索引
        user_key = f"user_sessions:{session.user_id}"
        await self.redis.sadd(user_key, session.session_id)
        await self.redis.expire(user_key, ttl)
    
    async def get(self, session_id: str) -> Optional[Session]:
        """获取会话"""
        key = f"session:{session_id}"
        data = await self.redis.get(key)
        
        if data:
            return Session.from_dict(json.loads(data))
        return None
    
    async def delete(self, session_id: str):
        """删除会话"""
        key = f"session:{session_id}"
        
        # 获取会话信息以清理索引
        session_data = await self.redis.get(key)
        if session_data:
            session = json.loads(session_data)
            user_id = session["user_id"]
            
            # 从用户索引中移除
            user_key = f"user_sessions:{user_id}"
            await self.redis.srem(user_key, session_id)
        
        await self.redis.delete(key)
    
    async def get_user_sessions(self, user_id: str) -> list:
        """获取用户所有会话"""
        user_key = f"user_sessions:{user_id}"
        session_ids = await self.redis.smembers(user_key)
        
        sessions = []
        for session_id in session_ids:
            session = await self.get(session_id)
            if session:
                sessions.append(session)
        
        return sessions
```

## 会话安全最佳实践

### 1. 会话 ID 生成

```python
import secrets
import hashlib

def generate_session_id() -> str:
    """生成安全的会话 ID"""
    # 使用加密安全的随机数生成器
    token = secrets.token_urlsafe(32)
    
    # 添加时间戳防止重放攻击
    timestamp = str(int(datetime.now().timestamp()))
    
    # 生成 HMAC
    signature = hashlib.sha256(
        f"{token}{timestamp}{SECRET_KEY}".encode()
    ).hexdigest()[:16]
    
    return f"{token}.{timestamp}.{signature}"

def verify_session_id(session_id: str) -> bool:
    """验证会话 ID 签名"""
    try:
        parts = session_id.split(".")
        if len(parts) != 3:
            return False
        
        token, timestamp, signature = parts
        
        # 验证签名
        expected_signature = hashlib.sha256(
            f"{token}{timestamp}{SECRET_KEY}".encode()
        ).hexdigest()[:16]
        
        return secrets.compare_digest(signature, expected_signature)
    except:
        return False
```

### 2. 会话固定攻击防护

```python
class SessionSecurity:
    @staticmethod
    async def regenerate_session_id(old_session_id: str, user_id: str) -> str:
        """重新生成会话 ID（防止会话固定攻击）"""
        
        # 1. 获取旧会话数据
        old_session = await session_store.get(old_session_id)
        
        # 2. 创建新会话
        new_session_id = generate_session_id()
        new_session = Session(
            session_id=new_session_id,
            user_id=user_id,
            messages=old_session.messages if old_session else [],
        )
        
        # 3. 保存新会话
        await session_store.save(new_session)
        
        # 4. 删除旧会话
        if old_session:
            await session_store.delete(old_session_id)
        
        return new_session_id
    
    @staticmethod
    async def on_login(user_id: str, old_session_id: Optional[str]):
        """登录时重新生成会话"""
        
        # 销毁旧会话，生成新会话
        if old_session_id:
            await SessionSecurity.regenerate_session_id(
                old_session_id, user_id
            )
```

### 3. 会话劫持防护

```python
class SessionValidator:
    def __init__(self):
        self.fingerprint_store = {}  # session_id -> fingerprint
    
    def generate_fingerprint(self, request: Request) -> str:
        """生成会话指纹"""
        user_agent = request.headers.get("User-Agent", "")
        accept_language = request.headers.get("Accept-Language", "")
        
        # 可选：使用客户端 IP（注意 NAT 环境）
        # client_ip = request.client.host
        
        fingerprint_data = f"{user_agent}|{accept_language}"
        return hashlib.sha256(fingerprint_data.encode()).hexdigest()
    
    async def validate_session(self, request: Request, session_id: str) -> bool:
        """验证会话指纹"""
        current_fingerprint = self.generate_fingerprint(request)
```

### 1. 会话数据压缩

```python
import gzip
import json

class CompressedSessionStore:
    async def save(self, session: Session):
        """压缩存储会话数据"""
        data = json.dumps(session.to_dict())
        compressed = gzip.compress(data.encode())
        
        await redis.setex(
            f"session:{session.session_id}",
            3600,
            compressed
        )
    
    async def get(self, session_id: str) -> Optional[Session]:
        """解压读取会话数据"""
        compressed = await redis.get(f"session:{session_id}")
        
        if compressed:
            data = gzip.decompress(compressed).decode()
            return Session.from_dict(json.loads(data))
        return None
```

### 2. 批量操作

```python
class BatchSessionStore:
    def __init__(self):
        self.write_buffer = {}  # session_id -> Session
        self.flush_interval = 5  # 5 秒刷写
    
    async def save_batch(self, sessions: list):
        """批量保存会话"""
        for session in sessions:
            self.write_buffer[session.session_id] = session
        
        # 定时刷写
        if len(self.write_buffer) >= 100:
            await self.flush()
    
    async def flush(self):
        """刷写缓冲到存储"""
        if not self.write_buffer:
            return
        
        # 批量写入 Redis
        pipe = redis.pipeline()
        
        for session_id, session in self.write_buffer.items():
            key = f"session:{session_id}"
            pipe.setex(key, 3600, json.dumps(session.to_dict()))
        
        await pipe.execute()
        self.write_buffer.clear()
```

### 3. 懒加载

```python
class LazySessionStore:
    def __init__(self):
        self.cache = {}  # session_id -> Session
        self.cache_ttl = 300  # 5 分钟缓存
    
    async def get(self, session_id: str) -> Optional[Session]:
        """懒加载：先查缓存，再查数据库"""
        
        # 检查缓存
        if session_id in self.cache:
            session, cached_at = self.cache[session_id]
            
            # 缓存未过期
            if datetime.now() - cached_at < timedelta(seconds=self.cache_ttl):
                return session
        
        # 缓存未命中，从数据库加载
        session = await db.get(session_id)
        
        if session:
            # 更新缓存
            self.cache[session_id] = (session, datetime.now())
        
        return session
```
